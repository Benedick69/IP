<body>
    <script>
        class Car{
                constructor(brand){
                    this.brand=brand;
                }
                present(){
                    console.log(`Car brand - ${this.brand}`)
                }
            }
        
    class Mod extends Car{
        constructor(brand,model){
            super(brand);
            this.model=model;
        }
        show(){
            console.log(`${this.brand}, It is a ${this.model}`)
        }
    }

    let mycar= new Mod("Porsche","911 turbo")
    mycar.show()

    </script>
